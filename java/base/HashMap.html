# HashMapеә•еұӮе®һзҺ°еҺҹзҗҶ

HashMapеҜ№иұЎеңЁе®һдҫӢеҢ–еҗҺпјҢеә•еұӮеҲӣе»әдәҶдёҖдёӘй•ҝеәҰдёә16зҡ„дёҖз»ҙж•°з»„ Entry[ ] table

еҪ“жү§иЎҢж–№жі• put(key1пјҢvalue1) ж·»еҠ ж•°жҚ®ж—¶пјҢHashMapеә•еұӮзҡ„е®һзҺ°еҺҹзҗҶеҰӮдёӢпјҲжӯӨеүҚеҸҜиғҪе·Із»Ҹжү§иЎҢиҝҮеӨҡж¬Ўput()ж–№жі•пјүпјҡ

йҰ–е…ҲпјҢи°ғз”Ёkey1жүҖеңЁзұ»зҡ„hashCode()и®Ўз®—key1зҡ„е“ҲеёҢеҖјпјҢжӯӨе“ҲеёҢеҖјз»ҸиҝҮеә•еұӮз®—жі•зҡ„и®Ўз®—д»ҘеҗҺпјҢеҸҜд»Ҙеҫ—еҲ°еңЁEntryж•°з»„дёӯзҡ„еӯҳж”ҫдҪҚзҪ®пјҢжҺҘдёӢжқҘеҸҲеҲҶдёәдёүз§Қжғ…еҶөгҖӮ

жғ…еҶө1пјҡеҰӮжһңжӯӨдҪҚзҪ®дёҠзҡ„ж•°жҚ®дёәз©әпјҢжӯӨж—¶зҡ„key1-value1ж·»еҠ жҲҗеҠҹгҖӮ

жғ…еҶө2пјҡеҰӮжһңжӯӨдҪҚзҪ®дёҠж•°жҚ®дёҚдёәз©әпјҲж„Ҹе‘ізқҖжӯӨдҪҚзҪ®дёҠеӯҳеңЁдёҖдёӘжҲ–еӨҡдёӘж•°жҚ®пјҲеӨҡдёӘж•°жҚ®д»Ҙй“ҫиЎЁзҡ„еҪўејҸеӯҳеңЁпјүпјүпјҢеҲҷжҜ”иҫғkey1е’Ңе·Із»ҸеӯҳеңЁзҡ„ж•°жҚ®зҡ„е“ҲеёҢеҖјпјҢеҰӮжһңkey1зҡ„е“ҲеёҢеҖјдёҺе·Із»ҸеӯҳеңЁзҡ„ж•°жҚ®зҡ„е“ҲеёҢеҖјйғҪдёҚзӣёеҗҢпјҢжӯӨж—¶key1-value1ж·»еҠ жҲҗеҠҹпјҢеҚіkey1-value1е’ҢеҺҹжқҘзҡ„ж•°жҚ®дјҡд»Ҙй“ҫиЎЁзҡ„еҪўејҸеӯҳеӮЁгҖӮ

жғ…еҶө3пјҡдёҺжғ…еҶөдәҢзұ»дјјпјҢеңЁжҜ”иҫғkey1е’Ңе·Із»ҸеӯҳеңЁзҡ„ж•°жҚ®зҡ„е“ҲеёҢеҖјж—¶пјҢеҰӮжһңkey1зҡ„е“ҲеёҢеҖје’ҢжҹҗдёӘе·Із»ҸеӯҳеңЁзҡ„ж•°жҚ®пјҲеҰӮkey2-value2пјүзҡ„е“ҲеёҢеҖјзӣёеҗҢпјҢеҲҷи°ғз”Ёkey1жүҖеңЁзұ»зҡ„equals()ж–№жі•пјҢиҝ”еӣһkey1е’Ңkey2зҡ„жҜ”иҫғз»“жһңпјҢеҰӮжһңиҝ”еӣһзҡ„жҳҜfalseпјҢжӯӨж—¶key1-value1ж·»еҠ жҲҗеҠҹпјҢеҗҢж ·д№ҹжҳҜе’ҢеҺҹжқҘзҡ„ж•°жҚ®д»Ҙй“ҫиЎЁзҡ„еҪўејҸеӯҳеӮЁпјӣеҰӮжһңиҝ”еӣһзҡ„жҳҜtrueпјҢеҲҷдјҡдҪҝз”Ёvalue1жӣҝжҚўvalue2гҖӮ

еңЁдёҚж–ӯж·»еҠ ж•°жҚ®зҡ„иҝҮзЁӢдёӯпјҢдјҡж¶үеҸҠеҲ°ж•°з»„зҡ„жү©е®№й—®йўҳгҖӮеҪ“ж·»еҠ ж•°жҚ®и¶…еҮәдёҙз•ҢеҖј(дё”ж•°жҚ®иҰҒеӯҳж”ҫзҡ„дҪҚзҪ®йқһз©ә)ж—¶пјҢйңҖиҰҒеҜ№ж•°з»„жү©е®№гҖӮй»ҳи®Өзҡ„жү©е®№ж–№ејҸпјҡжү©е®№дёәеҺҹжқҘе®№йҮҸзҡ„2еҖҚпјҢе№¶е°ҶеҺҹжқҘзҡ„ж•°жҚ®еӨҚеҲ¶иҝҮжқҘгҖӮ

java8еңЁз¬¬дәҢжӯҘд№ӢеүҚеӨҡдәҶдёҖжӯҘпјҢеҲӨж–ӯжҳҜеҗҰжҳҜж•°з»“жһ„гҖӮеҰӮжһңжҳҜз»“жһ„пјҢдҪҝз”Ёзәўй»‘ж ‘зҡ„putVlaueж–№жі•гҖӮ

еңЁз¬¬дәҢжӯҘж—¶д№ҹеӨҡдәҶдёҖдёӘеӨ„зҗҶгҖӮеҲӨиҜ»еҰӮжһңй“ҫиЎЁиҠӮзӮ№и¶…иҝҮ8еҸҳжҲҗзәўй»‘ж ‘з»“жһ„

/**
*
* @param hash keyжұӮhashзҡ„еҖј
* @param key зӣ®ж Үkey
* @param value зӣ®ж Үvalue
* @param onlyIfAbsent еҰӮжһңдёәзңҹпјҢеҲҷдёҚжӣҙж”№зҺ°жңүеҖјпјҢputеҮҪж•°дёўиҝҮжқҘзҡ„жҳҜfalse
* @param evict еҰӮжһңдёә falseпјҢеҲҷиЎЁеӨ„дәҺеҲӣе»әжЁЎејҸпјҢputеҮҪж•°дёўиҝҮжқҘtrue
* @return еүҚдёҖдёӘеҖјпјҢеҰӮжһңжІЎжңүпјҢеҲҷдёә null
*/
final V putVal(int hash, K key, V value, boolean onlyIfAbsent,
                boolean evict) {
    // жҺҘж”¶mapзҡ„иҠӮзӮ№ж•°з»„
    Node<K,V>[] tab; 
    // 
    Node<K,V> p; 
    //
    int n;
    // 
    int i;
    // жҠҠжҲҗе‘ҳеҸҳйҮҸиөӢеҖјз»ҷtabе№¶еҲӨж–ӯжҳҜеҗҰдёәnullпјӣеҰӮжһңдёҚдёәnullпјҢжҠҠиҠӮзӮ№ж•°з»„й•ҝеәҰиөӢеҖјз»ҷnе№¶еҲӨж–ӯй•ҝеәҰжҳҜеҗҰдёә0
    if ((tab = table) == null || (n = tab.length) == 0)
        // жқЎд»¶жҲҗз«ӢпјҢеҲқе§ӢеҢ–иҠӮзӮ№ж•°з»„гҖӮеҲҶеҲ«жҠҠеҲқе§ӢеҢ–еҗҺзҡ„иҠӮзӮ№ж•°з»„е’Ңй•ҝеәҰиөӢеҖјз»ҷtabгҖҒn
        n = (tab = resize()).length;

    // жңҖеҗҺиҠӮзӮ№дҪҚзҪ®е’Ңhashж•°жұӮдёҺеҖјиөӢеҖјз»ҷiпјҢд»ҺиҠӮзӮ№ж•°з»„дёӯеҸ–еҮәиҠӮзӮ№з»ҷpе№¶еҲӨж–ӯpжҳҜеҗҰдёәnull
    if ((p = tab[i = (n - 1) & hash]) == null)
        // жІЎжңүиҠӮзӮ№з”ҹжҲҗдёҖдёӘж–°зҡ„иҠӮзӮ№пјҢзӣҙжҺҘж”ҫе…Ҙ
        tab[i] = newNode(hash, key, value, null);
    else {
        Node<K,V> e; 
        K k;
        // иҠӮзӮ№зҡ„hashеҖјдёҺеҪ“еүҚkeyзҡ„hashеҖјжҜ”иҫғпјӣпјҲиҠӮзӮ№зҡ„keyеҖјиөӢеҖјз»ҷkпјҢдёҺзӣ®ж ҮkeyжҜ”иҫғпјү
        if (p.hash == hash &&
            ((k = p.key) == key || (key != null && key.equals(k))))
            // еҸ–еҮәиҠӮзӮ№з»ҷe
            e = p;
        else if (p instanceof TreeNode)
            // pжҳҜж•°з»“жһ„пјҢзӣҙжҺҘдҪҝз”Ёж•°з»“жһ„ж“ҚдҪңиҠӮзӮ№гҖӮeжҳҜputTreeValзҡ„иҝ”еӣһеҖј
            e = ((TreeNode<K,V>)p).putTreeVal(this, tab, hash, key, value);
        else {
            for (int binCount = 0; ; ++binCount) {
                // eжҳҜжңҖеҗҺдёҖдёӘз»“зӮ№
                if ((e = p.next) == null) {
                    p.next = newNode(hash, key, value, null);
                    // еҫӘзҺҜж¬Ўж•°и¶…иҝҮ7пјҢд»Һз¬¬8ж¬ЎејҖе§Ӣ
                    if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                        // ж”№дҪҝз”Ёзәўй»‘ж•°
                        treeifyBin(tab, hash);
                    break;
                }
                // дёӯй—ҙжңүkeyзӣёзӯү
                if (e.hash == hash &&
                    ((k = e.key) == key || (key != null && key.equals(k))))
                    break;
                // еҸ–еҲ°дёӢдёҖдёӘиҠӮзӮ№
                p = e;
            }
        }
        if (e != null) { // еӯҳеңЁkeyзҡ„жҳ е°„
            // иҖҒеҖј
            V oldValue = e.value;
            // жҳҜеҗҰжӣҝжҚўж–°еҖј
            if (!onlyIfAbsent || oldValue == null)
                e.value = value;
            afterNodeAccess(e);
            // иҝ”еӣһиҖҒеҖј
            return oldValue;
        }
    }
    // дҝ®ж”№и®Ўж•°еҠ 1
    ++modCount;
    // иҠӮзӮ№жҖ»ж•°еӨ§дәҺдёҙз•ҢеҖјпјҢйҮҚж–°жһ„е»ә
    if (++size > threshold)
        resize();
    afterNodeInsertion(evict);
    // иҝ”еӣһnull
    return null;
}

// mapпјҡзӣ®ж ҮдҪҝз”Ёзҡ„mapеҜ№иұЎ
// tabпјҡиҠӮзӮ№ж•°з»„
// hпјҡзӣ®ж Үkeyзҡ„hashеҖј
// kпјҡзӣ®ж Үkey
// vпјҡзӣ®ж Үvalue
final TreeNode<K,V> putTreeVal(HashMap<K,V> map, Node<K,V>[] tab,
                                int h, K k, V v) {
    // 
    Class<?> kc = null;
    // жҗңзҙўејҖе…і
    boolean searched = false;
    // жңүзҲ¶иҠӮзӮ№пјҢеҗ‘дёҠжүҫеҲ°ж №з»“зӮ№з»ҷrootпјӣжІЎжңүзҲ¶иҠӮзӮ№пјҢжң¬иҠӮзӮ№е°ұжҳҜж №иҠӮзӮ№
    TreeNode<K,V> root = (parent != null) ? root() : this;
    // д»Һж №иҠӮзӮ№ејҖе§ӢеҫӘзҺҜ
    for (TreeNode<K,V> p = root;;) {
        int dir, ph; K pk;
        if ((ph = p.hash) > h)
            // еҪ“еүҚиҠӮзӮ№зҡ„hsahеҖјжҜ”зӣ®ж Үkeyзҡ„hashеҖјеӨ§
            dir = -1;
        else if (ph < h)
            // еҪ“еүҚиҠӮзӮ№зҡ„hsahеҖјжҜ”зӣ®ж Үkeyзҡ„hashеҖје°Ҹ
            dir = 1;
        else if ((pk = p.key) == k || (k != null && k.equals(pk)))
            //еҪ“еүҚиҠӮзӮ№зҡ„hsahеҖјжҜ”зӣ®ж Үkeyзҡ„hashеҖјдёҖиҮҙпјҢдё”keyдёҖиҮҙгҖӮиҝ”еӣһеҪ“еүҚиҠӮзӮ№
            return p;
        else if ((kc == null &&
                    (kc = comparableClassFor(k)) == null) ||
                    (dir = compareComparables(kc, k, pk)) == 0) {
            // еҪ“еүҚиҠӮзӮ№зҡ„hsahеҖјжҜ”зӣ®ж Үkeyзҡ„hashеҖјдёҖиҮҙпјҢдё”keyдёҚдёҖиҮҙгҖӮ
            // зӣ®ж ҮkeyдёҚеҸҜжҜ”иҫғ жҲ– еҸҜжҜ”иҫғдҪҶжҜ”иҫғз»“жһңзӣёзӯү
            if (!searched) {
                // жІЎжңүжҗңзҙўиҝҮ
                TreeNode<K,V> q, ch;
                // и®ҫдёәжҗңзҙўиҝҮ
                searched = true;
                // еҪ“еүҚиҠӮзӮ№зҡ„е·Ұеӯ©еӯҗдёҚдёәз©әпјҢе·Ұиҫ№иғҪжүҫеҲ°пјҢжүҫеҲ°иҠӮзӮ№иөӢеҖјз»ҷq
                // жҲ–иҖ…
                // еҸіиҫ№дёҚдёәз©әпјҢеҸіиҫ№иғҪжүҫеҲ°пјҢжүҫеҲ°иҠӮзӮ№иөӢеҖјз»ҷq
                if (((ch = p.left) != null &&
                        (q = ch.find(h, k, kc)) != null) ||
                    ((ch = p.right) != null &&
                        (q = ch.find(h, k, kc)) != null))
                    // жүҫеҲ°зӣҙжҺҘиҝ”еӣһ
                    return q;
            }
            // зӣ®ж ҮkeyдёҚеҸҜжҜ”иҫғ жҲ– еҸҜжҜ”иҫғдҪҶжҜ”иҫғз»“жһңзӣёзӯү зҡ„и§ЈеҶіж–№жі•гҖӮ
            // и§ЈеҶіеҠһжі•пјҡеҸӘйңҖиҰҒдёҖиҮҙзҡ„жҸ’е…Ҙи§„еҲҷжқҘдҝқжҢҒйҮҚж–°е№іиЎЎд№Ӣй—ҙзҡ„зӯүд»·жҖ§
            // зұ»еҗҚжҜ”иҫғеҫ—еҲ°з»“жһңгҖӮ
            // зұ»еҗҚиҝҳдёҖиҮҙйҖҡиҝҮ:(System.identityHashCode(k) <= System.identityHashCode(pk) ?-1 : 1)
            dir = tieBreakOrder(k, pk);
        }

        // еҪ“еүҚиҠӮзӮ№иөӢеҖјз»ҷxp
        TreeNode<K,V> xp = p;
        // p еҺ»еҲ°е·ҰиҠӮзӮ№жҲ–еҸіиҠӮзӮ№гҖӮ
        if ((p = (dir <= 0) ? p.left : p.right) == null) {
            // иҠӮзӮ№дёәnullпјҢxpзҡ„nextиҠӮзӮ№иөӢеҖјз»ҷxpn
            Node<K,V> xpn = xp.next;
            // ж–°е»әдёҖдёӘз»“зӮ№x
            TreeNode<K,V> x = map.newTreeNode(h, k, v, xpn);
            if (dir <= 0)
                // dirе°ҸдәҺ0пјҢж”ҫеңЁе·ҰиҠӮзӮ№
                xp.left = x;
            else
                // еӨ§дәҺ0пјҢж”ҫеңЁеҸіиҠӮзӮ№
                xp.right = x;
            // xpзҡ„naxtжҢҮеҗ‘ж–°иҠӮзӮ№
            xp.next = x;
            // xзҡ„зҲ¶иҠӮзӮ№жҢҮеҗ‘xзҡ„еүҚдёҖдёӘз»“зӮ№
            // xзҡ„еүҚдёҖдёӘз»“зӮ№жҢҮеҗ‘xp
            x.parent = x.prev = xp;
            if (xpn != null)
                ((TreeNode<K,V>)xpn).prev = x;
            // еҒҡи°ғж•ҙйҮҚж–°жүҫеҲ°rootпјҢдҝ®ж”№иҠӮзӮ№ж•°з»„гҖӮзЎ®дҝқз»ҷе®ҡзҡ„ж №жҳҜе…¶ bin зҡ„з¬¬дёҖдёӘиҠӮзӮ№гҖӮ
            moveRootToFront(tab, balanceInsertion(root, x));
            return null;
        }
    }
}

// еҲқе§ӢеҢ–жҲ–еҠ еҖҚиЎЁеӨ§е°ҸгҖӮеҰӮжһңдёәз©әпјҢеҲҷжҢүз…§еӯ—ж®өйҳҲеҖјдёӯдҝқеӯҳзҡ„еҲқе§Ӣе®№йҮҸзӣ®ж ҮиҝӣиЎҢеҲҶй…ҚгҖӮ
// еҗҰеҲҷпјҢеӣ дёәжҲ‘д»¬дҪҝз”ЁдәҢж¬Ўе№Ӯеұ•ејҖпјҢжҜҸдёӘ bin дёӯзҡ„е…ғзҙ еҝ…йЎ»дҝқжҢҒзӣёеҗҢзҡ„зҙўеј•пјҢжҲ–иҖ…еңЁж–°иЎЁдёӯд»ҘдәҢж¬Ўе№Ӯзҡ„еҒҸз§»йҮҸз§»еҠЁгҖӮ
final Node<K,V>[] resize() {
    Node<K,V>[] oldTab = table;
    int oldCap = (oldTab == null) ? 0 : oldTab.length;
    int oldThr = threshold;
    int newCap, newThr = 0;
    if (oldCap > 0) {
        if (oldCap >= MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return oldTab;
        }
        else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                    oldCap >= DEFAULT_INITIAL_CAPACITY)
            newThr = oldThr << 1; // double threshold
    }
    else if (oldThr > 0) // initial capacity was placed in threshold
        newCap = oldThr;
    else {               // zero initial threshold signifies using defaults
        newCap = DEFAULT_INITIAL_CAPACITY;
        newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
    }
    if (newThr == 0) {
        float ft = (float)newCap * loadFactor;
        newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                    (int)ft : Integer.MAX_VALUE);
    }
    threshold = newThr;
    @SuppressWarnings({"rawtypes","unchecked"})
        Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
    table = newTab;
    if (oldTab != null) {
        for (int j = 0; j < oldCap; ++j) {
            Node<K,V> e;
            if ((e = oldTab[j]) != null) {
                oldTab[j] = null;
                if (e.next == null)
                    newTab[e.hash & (newCap - 1)] = e;
                else if (e instanceof TreeNode)
                    ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                else { // preserve order
                    Node<K,V> loHead = null, loTail = null;
                    Node<K,V> hiHead = null, hiTail = null;
                    Node<K,V> next;
                    do {
                        next = e.next;
                        if ((e.hash & oldCap) == 0) {
                            if (loTail == null)
                                loHead = e;
                            else
                                loTail.next = e;
                            loTail = e;
                        }
                        else {
                            if (hiTail == null)
                                hiHead = e;
                            else
                                hiTail.next = e;
                            hiTail = e;
                        }
                    } while ((e = next) != null);
                    if (loTail != null) {
                        loTail.next = null;
                        newTab[j] = loHead;
                    }
                    if (hiTail != null) {
                        hiTail.next = null;
                        newTab[j + oldCap] = hiHead;
                    }
                }
            }
        }
    }
    return newTab;
}
